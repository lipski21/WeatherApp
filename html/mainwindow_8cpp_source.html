193void WizualizacjaPogody::updateTempChart(const QJsonArray &temperatures, const QJsonArray &timeArray){
194
195 QLineSeries *series = new QLineSeries();
196
197 for (int i = 0; i < temperatures.size(); ++i) {
198 QDateTime dateTime = QDateTime::fromString(timeArray[i].toString(), Qt::ISODate);
199 double temperature = temperatures[i].toDouble();
200 series->append(dateTime.toMSecsSinceEpoch(), temperature);
201 }
202
203 QChart *chart = new QChart();
204 chart->legend()->hide();
205 chart->addSeries(series);
206 chart->setTitle("Zmiany temperaturowe");
207
208
209 // Ustawienia osi X (czas)
210 QDateTime minDateTime = QDateTime::fromString(timeArray.first().toString(), Qt::ISODate);
211 QDateTime maxDateTime = QDateTime::fromString(timeArray.last().toString(), Qt::ISODate);
212 QDateTimeAxis *axisX = new QDateTimeAxis;
213 axisX->setTickCount(8);
214 axisX->setFormat("MM-dd hh:mm");
215 axisX->setTitleText("Data");
216 axisX->setRange(minDateTime, maxDateTime);
217 chart->addAxis(axisX, Qt::AlignBottom);
218 series->attachAxis(axisX);
219
220 // Ustawienia osi Y (temperatura)
221 double minTemperature = std::numeric_limits<double>::max();
222 double maxTemperature = std::numeric_limits<double>::min();
223 for (const QJsonValue &value : temperatures) {
224 double temperature = value.toDouble();
225 minTemperature = qMin(minTemperature, temperature);
226 maxTemperature = qMax(maxTemperature, temperature);
227 }
228 QValueAxis *axisY = new QValueAxis;
229 axisY->setLabelFormat("%.1f");
230 axisY->setTitleText("Temperatura (°C)");
231 axisY->setRange(minTemperature - 1, maxTemperature + 1); // Rozszerz zakres o 1 stopień w obu kierunkach
232 chart->addAxis(axisY, Qt::AlignLeft);
233 series->attachAxis(axisY);
234
235 // Tworzenie widoku wykresu
236 TempChart->setChart(chart);
237 QSize newSize = TempChart->parentWidget()->size();
238 chart->resize(newSize);
239 TempChart->update();
240}
